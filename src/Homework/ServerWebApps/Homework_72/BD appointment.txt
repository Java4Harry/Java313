USE system_fitness;

CREATE TABLE IF NOT EXISTS appointment(
id INT PRIMARY KEY AUTO_INCREMENT,
client_id INT NOT NULL,
fullname VARCHAR(45) NOT NULL,
gender VARCHAR(45) NOT NULL,
age VARCHAR(45) NOT NULL,
appoint_date VARCHAR(45) NOT NULL,
email VARCHAR(45) NOT NULL,
phone VARCHAR(45) NOT NULL,
likes VARCHAR(45) NOT NULL,
worker_id INT NOT NULL,
address VARCHAR(45) NOT NULL,
status VARCHAR(45) NOT NULL,
FOREIGN KEY(client_id) REFERENCES client_data(id) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY(worker_id) REFERENCES worker(id) ON DELETE CASCADE ON UPDATE CASCADE
);
